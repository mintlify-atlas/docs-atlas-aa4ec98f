# LLM Tracing and Observability

MLflow Tracing provides comprehensive observability for LLM applications by capturing detailed execution traces. Each trace contains a hierarchical structure of spans that represent function calls, LLM invocations, and other operations.

## Why Tracing?

Tracing helps you:

- **Debug** complex LLM chains and agent workflows
- **Monitor** production LLM applications for performance and quality
- **Evaluate** LLM outputs by analyzing captured traces
- **Optimize** token usage and latency bottlenecks
- **Understand** multi-step reasoning and tool usage

## Quick Start

### Enable Tracing

```python
import mlflow

# Enable tracing globally
mlflow.tracing.enable()

# Now all supported frameworks are automatically traced
import openai

client = openai.OpenAI()
response = client.chat.completions.create(
    model="gpt-4o-mini",
    messages=[{"role": "user", "content": "Hello!"}]
)

# Access the last trace
trace_id = mlflow.get_last_active_trace_id()
trace = mlflow.get_trace(trace_id)
```

### Manual Tracing with Decorator

```python
import mlflow

@mlflow.trace
def process_user_query(question: str) -> str:
    # Automatically captures inputs, outputs, and exceptions
    context = retrieve_context(question)
    answer = generate_answer(question, context)
    return answer

@mlflow.trace(span_type="RETRIEVER")
def retrieve_context(query: str) -> list[str]:
    # Nested function calls create child spans
    return ["context 1", "context 2"]

@mlflow.trace(span_type="LLM")
def generate_answer(question: str, context: list[str]) -> str:
    import openai
    client = openai.OpenAI()
    
    prompt = f"Context: {context}\nQuestion: {question}"
    response = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[{"role": "user", "content": prompt}]
    )
    return response.choices[0].message.content

# Call the function - creates a trace with 3 spans
result = process_user_query("What is MLflow?")
```

## Trace Structure

A trace consists of:

- **Trace**: Top-level container with metadata (ID, timestamp, status)
- **Spans**: Individual operations forming a tree structure
  - **Root span**: Entry point of the trace
  - **Child spans**: Nested operations

```python
# Example trace structure
trace = mlflow.get_trace(trace_id)

print(f"Trace ID: {trace.info.trace_id}")
print(f"Status: {trace.info.status}")
print(f"Duration: {trace.info.execution_duration} ms")

# Access spans
for span in trace.data.spans:
    print(f"Span: {span.name}")
    print(f"  Type: {span.span_type}")
    print(f"  Inputs: {span.inputs}")
    print(f"  Outputs: {span.outputs}")
    print(f"  Duration: {span.duration} ms")
```

MLflow automatically traces calls to supported frameworks:


### 2. Decorator-based Tracing

Use `@mlflow.trace` for custom functions:

```python
import mlflow

@mlflow.trace
def my_function(x: int) -> int:
    return x + 1

# With custom span type and attributes
@mlflow.trace(
    span_type="PARSER",
    attributes={"parser_version": "1.0"}
)
def parse_document(doc: str) -> dict:
    # Your parsing logic
    return {"title": "...", "content": "..."}
```

### 3. Context Manager Tracing

Use `mlflow.start_span()` for fine-grained control:

```python
import mlflow

with mlflow.start_span(name="data_processing") as span:
    # Set inputs explicitly
    span.set_inputs({"batch_size": 100})
    
    # Process data
    result = process_batch(data)
    
    # Set outputs and attributes
    span.set_outputs({"processed_count": len(result)})
    span.set_attribute("processing_time", 1.5)
```

### 4. Low-level API

For complete control over span lifecycle:

```python
import mlflow

# Start a root span
root_span = mlflow.start_span_no_context(
    name="root_operation",
    span_type="CHAIN",
    inputs={"user_id": "123"}
)

# Create child spans
child_span = mlflow.start_span_no_context(
    name="child_operation",
    parent_span=root_span,
    span_type="LLM"
)

# Do work...
child_span.end(outputs={"result": "..."})
root_span.end()
```

## Capturing Metadata

### Setting Span Inputs and Outputs

```python
@mlflow.trace
def process_query(question: str) -> str:
    # Inputs/outputs are captured automatically
    return generate_answer(question)

# Or set explicitly with context manager
with mlflow.start_span("custom_span") as span:
    span.set_inputs({"query": "What is MLflow?"})
    result = do_work()
    span.set_outputs({"answer": result})
```

### Setting Span Attributes

```python
with mlflow.start_span("llm_call") as span:
    span.set_attribute("model", "gpt-4o")
    span.set_attribute("temperature", 0.7)
    span.set_attribute("max_tokens", 1000)
    
    # Or set multiple at once
    span.set_attributes({
        "provider": "openai",
        "request_id": "req_123"
    })
```

### Recording Events

```python
from mlflow.entities.span_event import SpanEvent

with mlflow.start_span("processing") as span:
    # Record custom events
    span.add_event(SpanEvent(
        name="cache_miss",
        attributes={"key": "user_123"}
    ))
    
    # Record exceptions
    try:
        risky_operation()
    except Exception as e:
        span.add_event(SpanEvent.from_exception(e))
        raise
```

## Streaming and Async Support

### Generator Functions

```python
@mlflow.trace
def stream_tokens(prompt: str):
    """Stream tokens from LLM"""
    client = openai.OpenAI()
    stream = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[{"role": "user", "content": prompt}],
        stream=True
    )
    
    for chunk in stream:
        if chunk.choices[0].delta.content:
            yield chunk.choices[0].delta.content

# Streaming is automatically captured
for token in stream_tokens("Tell me a story"):
    print(token, end="")
```

### Custom Output Reducer

```python
def concat_chunks(chunks: list[str]) -> str:
    """Combine streamed chunks into final output"""
    return "".join(chunks)

@mlflow.trace(output_reducer=concat_chunks)
def stream_response(prompt: str):
    for chunk in generate_stream(prompt):
        yield chunk
```

```python
import asyncio
```

## Distributed Tracing

### Context Propagation

Propagate trace context across service boundaries:

```python
import mlflow
import requests

@mlflow.trace
def call_remote_service():
    # Get context headers to propagate
    headers = mlflow.tracing.get_tracing_context_headers_for_http_request()
    
    # Send request with trace context
    response = requests.post(
        "https://api.example.com/predict",
        headers=headers,
        json={"input": "data"}
    )
    return response.json()

# On the receiving service
from flask import Flask, request

app = Flask(__name__)

@app.route("/predict", methods=["POST"])
def predict():
    # Extract trace context from headers
    mlflow.tracing.set_tracing_context_from_http_request_headers(
        request.headers
    )
    
    # Spans created here are part of the distributed trace
    with mlflow.start_span("prediction"):
        result = model.predict(request.json["input"])
        return {"result": result}
```

### Multi-threading

```python
import mlflow
from concurrent.futures import ThreadPoolExecutor

@mlflow.trace
def parent_function():
    # Get current span to propagate to threads
    current_span = mlflow.get_current_active_span()
    
    def worker(i):
        # Create child span in thread
        child_span = mlflow.start_span_no_context(
            name=f"worker_{i}",
            parent_span=current_span
        )
        # Do work...
        child_span.end(outputs={"result": i})
    
    with ThreadPoolExecutor(max_workers=4) as executor:
        executor.map(worker, range(10))
```

### Agent with Tool Calls

```python
import mlflow
import openai
from typing import Any, Dict

@mlflow.trace(span_type="AGENT")
def run_agent(user_query: str) -> str:
    """Agent that can use tools"""
    client = openai.OpenAI()
    
    messages = [{"role": "user", "content": user_query}]
    
    tools = [
        {
            "type": "function",
            "function": {
                "name": "get_weather",
                "description": "Get weather for a location",
                "parameters": {
                    "type": "object",
                    "properties": {"location": {"type": "string"}}
                }
            }
        }
    ]
    
    response = client.chat.completions.create(
        model="gpt-4o",
        messages=messages,
        tools=tools
    )
    
    # Handle tool calls
    if response.choices[0].message.tool_calls:
        for tool_call in response.choices[0].message.tool_calls:
            result = execute_tool(tool_call)
            messages.append({
                "role": "tool",
                "content": str(result),
                "tool_call_id": tool_call.id
            })
        
        # Get final response
        final_response = client.chat.completions.create(
            model="gpt-4o",
            messages=messages
        )
        return final_response.choices[0].message.content
    
    return response.choices[0].message.content

@mlflow.trace(span_type="TOOL")
def execute_tool(tool_call: Any) -> Dict[str, Any]:
    """Execute a tool call"""
    if tool_call.function.name == "get_weather":
        return {"temperature": 72, "condition": "sunny"}
    return {}

# Run agent
result = run_agent("What's the weather in San Francisco?")
```
