## The Production Workflow

A typical production workflow involves:

1. **Train and log models** during experimentation
2. **Register promising models** in the Model Registry
3. **Transition models through stages** (Staging → Production)
4. **Deploy and serve models** in production
5. **Monitor and update** as needed

## Step 1: Train and Log Your Model

First, train a model and log it with MLflow:

<Steps>

<Step title="Train your model">

```python
import pandas as pd
from sklearn.datasets import load_iris
from sklearn.linear_model import ElasticNet
from sklearn.model_selection import train_test_split
from sklearn.metrics import mean_squared_error, mean_absolute_error, r2_score
import mlflow
import mlflow.sklearn

# Load and prepare data
iris = load_iris()
iris_df = pd.DataFrame(data=iris.data, columns=iris.feature_names)
iris_df["quality"] = (iris.target == 2).astype(int)

# Split data
train_df, test_df = train_test_split(iris_df, test_size=0.2, random_state=42)

# Prepare features and target
train_x = train_df.drop(["quality"], axis=1)
train_y = train_df[["quality"]]
test_x = test_df.drop(["quality"], axis=1)
test_y = test_df[["quality"]]
```

</Step>

<Step title="Log the model with MLflow">

```python
with mlflow.start_run(run_name="production_candidate") as training_run:
    # Register training dataset
    train_dataset = mlflow.data.from_pandas(train_df, name="train")
    
    # Train model
    lr = ElasticNet(alpha=0.5, l1_ratio=0.5, random_state=42)
    lr.fit(train_x, train_y)
    
    # Log model with parameters
    logged_model = mlflow.sklearn.log_model(
        sk_model=lr,
        name="elasticnet",
        params={
            "alpha": 0.5,
            "l1_ratio": 0.5,
        },
        input_example=train_x.head(),
    )
    
    # Evaluate and log metrics
    predictions = lr.predict(train_x)
    rmse = mean_squared_error(train_y, predictions, squared=False)
    mae = mean_absolute_error(train_y, predictions)
    r2 = r2_score(train_y, predictions)
    
    mlflow.log_metrics(
        metrics={"rmse": rmse, "r2": r2, "mae": mae},
        model_id=logged_model.model_id,
        dataset=train_dataset,
    )
    
    print(f"Model ID: {logged_model.model_id}")
    print(f"Model URI: {logged_model.model_uri}")
    print(f"RMSE: {rmse:.4f}, R2: {r2:.4f}, MAE: {mae:.4f}")
```

</Step>

</Steps>

## Step 2: Register the Model

Once you have a model you want to promote, register it in the Model Registry:

### Option A: Register During Training

You can register a model directly when logging it:

```python
with mlflow.start_run():
    model = LinearRegression().fit([[1], [2]], [3, 4])
    
    # Log and register in one step
    model_info = mlflow.sklearn.log_model(
        model,
        name="model",
        registered_model_name="iris_classifier"
    )
    
    print(f"Registered model version: {model_info.registered_model_version}")
```

### Option B: Register After Training

Register a previously logged model:

```python
import mlflow

# Register the model
model_name = "iris_production_model"
model_version = mlflow.register_model(
    model_uri=logged_model.model_uri,
    name=model_name
)

print(f"Registered model '{model_name}' version {model_version.version}")
```

### Using the Client API

For more control, use the MlflowClient:

```python
from mlflow.tracking import MlflowClient

client = MlflowClient()

# Create a new registered model
try:
    client.create_registered_model("iris_production_model")
    print("Created new registered model")
except mlflow.exceptions.MlflowException:
    print("Model already exists")

# Create a model version
model_version = client.create_model_version(
    name="iris_production_model",
    source=logged_model.model_uri,
    model_id=logged_model.model_id
)

print(f"Created version {model_version.version}")
```

## Step 3: Manage Model Stages

MLflow Model Registry supports model lifecycle stages:

- **None**: Initial stage
- **Staging**: Testing in a staging environment
- **Production**: Deployed to production
- **Archived**: No longer active

### Transition Models Between Stages

```python
from mlflow.tracking import MlflowClient

client = MlflowClient()
model_name = "iris_production_model"

# Move to Staging for testing
client.transition_model_version_stage(
    name=model_name,
    version=1,
    stage="Staging"
)
print(f"Model version 1 moved to Staging")

# After validation, move to Production
client.transition_model_version_stage(
    name=model_name,
    version=1,
    stage="Production"
)
print(f"Model version 1 moved to Production")

# Archive old version when deploying new one
client.transition_model_version_stage(
    name=model_name,
    version=1,
    stage="Archived"
)
print(f"Model version 1 archived")
```

### Add Descriptions and Tags

Document your models:

```python
# Update model version description
client.update_model_version(
    name=model_name,
    version=1,
    description="ElasticNet model with alpha=0.5, l1_ratio=0.5. Trained on 2024-03-05. Test accuracy: 0.95"
)

# Set tags for easy filtering
client.set_model_version_tag(
    name=model_name,
    version=1,
    key="validation_status",
    value="approved"
)

client.set_model_version_tag(
    name=model_name,
    version=1,
    key="deployment_date",
    value="2024-03-05"
)
```

## Step 4: Load and Serve Models

### Loading Models by Stage

Load the current production model:

```python
import mlflow.sklearn

model_name = "iris_production_model"

# Load the latest Production version
model_uri = f"models:/{model_name}/Production"
model = mlflow.sklearn.load_model(model_uri)

# Make predictions
import numpy as np
sample_input = np.array([[5.1, 3.5, 1.4, 0.2]])
prediction = model.predict(sample_input)

print(f"Prediction: {prediction}")
```

### Loading Models by Version

Load a specific version:


Use the model ID for precise tracking:

### Building a Prediction Service

Create a production-ready prediction service:

```python
import mlflow.sklearn
from flask import Flask, request, jsonify
import pandas as pd

app = Flask(__name__)

# Load model at startup
model_name = "iris_production_model"
model = mlflow.sklearn.load_model(f"models:/{model_name}/Production")

@app.route('/predict', methods=['POST'])
def predict():
    try:
        # Parse input
        data = request.get_json()
        input_df = pd.DataFrame(data['instances'])
        
        # Make prediction
        predictions = model.predict(input_df)
        
        # Return results
        return jsonify({
            'predictions': predictions.tolist()
        })
    except Exception as e:
        return jsonify({'error': str(e)}), 400

@app.route('/health', methods=['GET'])
def health():
    return jsonify({'status': 'healthy', 'model': model_name})

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=8080)
```

## Step 6: Deployment Best Practices

### A/B Testing New Models

Gradually roll out new model versions:

```python
import random
import mlflow.sklearn

# Load both models
model_a = mlflow.sklearn.load_model("models:/iris_production_model/1")
model_b = mlflow.sklearn.load_model("models:/iris_production_model/2")

def predict_with_ab_test(input_data, rollout_percentage=20):
    """
    Route traffic between model versions.
    
    Args:
        input_data: Input features
        rollout_percentage: Percentage of traffic for model B (0-100)
    """
    # Randomly select model based on rollout percentage
    if random.randint(1, 100) <= rollout_percentage:
        model = model_b
        version = "v2"
    else:
        model = model_a
        version = "v1"
    
    prediction = model.predict(input_data)
    
    # Log which version was used for monitoring
    print(f"Using model {version}")
    
    return prediction
```

### Model Validation Before Production

Validate models before promoting to production:

```python
def validate_model(model_name, version, test_data_x, test_data_y, threshold=0.90):
    """
    Validate model meets performance requirements.
    """
    # Load model
    model_uri = f"models:/{model_name}/{version}"
    model = mlflow.sklearn.load_model(model_uri)
    
    # Evaluate
    from sklearn.metrics import accuracy_score
    predictions = model.predict(test_data_x)
    accuracy = accuracy_score(test_data_y, predictions)
    
    # Check threshold
    if accuracy >= threshold:
        print(f"✓ Model validation passed: accuracy={accuracy:.4f}")
        return True
    else:
        print(f"✗ Model validation failed: accuracy={accuracy:.4f} < {threshold}")
        return False

# Validate before promoting
if validate_model("iris_production_model", 2, test_x, test_y):
    client.transition_model_version_stage(
        name="iris_production_model",
        version=2,
        stage="Production"
    )
```

### Monitoring Model Performance

Track model performance in production:

```python
import mlflow
from datetime import datetime

def log_prediction_metrics(model_name, version, actual, predicted):
    """
    Log prediction metrics for monitoring.
    """
    from sklearn.metrics import accuracy_score, f1_score
    
    with mlflow.start_run(run_name=f"production_monitoring_{datetime.now().strftime('%Y%m%d')}"):
        # Log model info
        mlflow.set_tag("model_name", model_name)
        mlflow.set_tag("model_version", version)
        mlflow.set_tag("environment", "production")
        
        # Calculate and log metrics
        accuracy = accuracy_score(actual, predicted)
        f1 = f1_score(actual, predicted, average='weighted')
        
        mlflow.log_metric("production_accuracy", accuracy)
        mlflow.log_metric("production_f1", f1)
        
        print(f"Logged production metrics: accuracy={accuracy:.4f}, f1={f1:.4f}")
```

## Complete Production Workflow Example

Here's a complete example bringing it all together:

```python
import mlflow
import mlflow.sklearn
from mlflow.tracking import MlflowClient
from sklearn.linear_model import LogisticRegression
from sklearn.datasets import load_iris
from sklearn.model_selection import train_test_split
from sklearn.metrics import accuracy_score

# 1. Train model
X, y = load_iris(return_X_y=True)
X_train, X_test, y_train, y_test = train_test_split(X, y, random_state=42)

with mlflow.start_run(run_name="production_pipeline"):
    # Train
    model = LogisticRegression(max_iter=200)
    model.fit(X_train, y_train)
    
    # Evaluate
    test_accuracy = accuracy_score(y_test, model.predict(X_test))
    mlflow.log_metric("test_accuracy", test_accuracy)
    
    # Log model with registration
    model_info = mlflow.sklearn.log_model(
        model,
        name="model",
        registered_model_name="production_iris_classifier"
    )
    
    model_name = "production_iris_classifier"
    version = model_info.registered_model_version

# 2. Validate model
client = MlflowClient()
if test_accuracy >= 0.90:
    # 3. Move to staging
    client.transition_model_version_stage(
        name=model_name,
        version=version,
        stage="Staging"
    )
    print(f"Model {model_name} v{version} moved to Staging")
    
    # 4. After staging tests pass, move to production
    client.transition_model_version_stage(
        name=model_name,
        version=version,
        stage="Production"
    )
    print(f"Model {model_name} v{version} deployed to Production")
else:
    print(f"Model accuracy {test_accuracy:.4f} below threshold, not promoting")

# 5. Load and use production model
production_model = mlflow.sklearn.load_model(f"models:/{model_name}/Production")
print(f"Loaded production model, ready to serve")
```
